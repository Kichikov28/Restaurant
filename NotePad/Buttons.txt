private void buttonAddToCart_Click(object sender, EventArgs e)
        {
            // Add the selected item to the cart
            if (listBoxMenu.SelectedItem != null)
            {
                listBoxCart.Items.Add(listBoxMenu.SelectedItem);
            }
        }


private void buttonRemoveFromCart_Click(object sender, EventArgs e)
        {
            // Remove the selected item from the cart
            if (listBoxCart.SelectedItem != null)
            {
                listBoxCart.Items.Remove(listBoxCart.SelectedItem);
            }
        }
